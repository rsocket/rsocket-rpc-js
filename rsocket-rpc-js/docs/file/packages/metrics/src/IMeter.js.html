'use strict';

import RawMeterTag from './RawMeterTag';
import {Meter} from './proto/metrics_pb';

/**
 */
export interface IMeter {
  name: string;
  description: ?string;
  statistic: string;
  type: string;
  tags: RawMeterTag[];
  units?: string;

  /**
   * @abstract
   * @returns {Object}
   */
  rates(): Object;

  /**
   * @abstract
   * @param {function} converter
   * @returns {Meter[]}
   */
  convert(converter: (IMeter) => Meter[]): Meter[];
}
